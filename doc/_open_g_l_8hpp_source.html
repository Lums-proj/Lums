36  inline Matrix4f
37  frustum(GLfloat left, GLfloat right, GLfloat bottom, GLfloat top, GLfloat znear, GLfloat zfar)
38  {
39  Matrix4f matrix;
40 
41  matrix[0][0] = 2.f * znear / (right - left);
42  matrix[1][1] = 2.f * znear / (top - bottom);
43  matrix[2][0] = (right + left) / (right - left);
44  matrix[2][1] = (top + bottom) / (top - bottom);
45  matrix[2][2] = -(zfar + znear) / (zfar - znear);
46  matrix[2][3] = -1.f;
47  matrix[3][2] = -(2.f * zfar * znear) / (zfar - znear);
48  return matrix;
49  }

90  inline void
91  rotate(Matrix4f& matrix, float angle, Vector3f v)
92  {
93  angle = angle * LUMS_PI / 180.f;
94  const float c = std::cos(angle);
95  const float s = std::sin(angle);
96 
97  Vector3f axis(normal(v));
98  Vector3f temp(axis * (1.f - c));
99 
100  Matrix4f rot;
101 
102  rot[0][0] = c + temp[0] * axis[0];
103  rot[0][1] = 0 + temp[0] * axis[1] + s * axis[2];
104  rot[0][2] = 0 + temp[0] * axis[2] - s * axis[1];
105 
106  rot[1][0] = 0 + temp[1] * axis[0] - s * axis[2];
107  rot[1][1] = c + temp[1] * axis[1];
108  rot[1][2] = 0 + temp[1] * axis[2] + s * axis[0];
109 
110  rot[2][0] = 0 + temp[2] * axis[0] + s * axis[1];
111  rot[2][1] = 0 + temp[2] * axis[1] - s * axis[0];
112  rot[2][2] = c + temp[2] * axis[2];
113 
114  rot[3][3] = 1.f;
115 
116  matrix *= rot;
117  }
