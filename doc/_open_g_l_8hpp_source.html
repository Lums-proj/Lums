47  frustum(GLfloat left, GLfloat right, GLfloat bottom, GLfloat top, GLfloat znear, GLfloat zfar)
48  {
49  Matrix4f matrix;
50 
51  matrix[0][0] = 2.f * znear / (right - left);
52  matrix[1][1] = 2.f * znear / (top - bottom);
53  matrix[2][0] = (right + left) / (right - left);
54  matrix[2][1] = (top + bottom) / (top - bottom);
55  matrix[2][2] = -(zfar + znear) / (zfar - znear);
56  matrix[2][3] = -1.f;
57  matrix[3][2] = -(2.f * zfar * znear) / (zfar - znear);
58  return matrix;
59  }

135  rotate(Matrix4f& matrix, float angle, Vector3f v)
136  {
137  angle = angle * LUMS_PI / 180.f;
138  const float c = std::cos(angle);
139  const float s = std::sin(angle);
140 
141  Vector3f axis(normal(v));
142  Vector3f temp(axis * (1.f - c));
143 
144  Matrix4f rot;
145 
146  rot[0][0] = c + temp[0] * axis[0];
147  rot[0][1] = 0 + temp[0] * axis[1] + s * axis[2];
148  rot[0][2] = 0 + temp[0] * axis[2] - s * axis[1];
149 
150  rot[1][0] = 0 + temp[1] * axis[0] - s * axis[2];
151  rot[1][1] = c + temp[1] * axis[1];
152  rot[1][2] = 0 + temp[1] * axis[2] + s * axis[0];
153 
154  rot[2][0] = 0 + temp[2] * axis[0] + s * axis[1];
155  rot[2][1] = 0 + temp[2] * axis[1] - s * axis[0];
156  rot[2][2] = c + temp[2] * axis[2];
157
